/**
 * Scrolls to a specific Element
 * @param {String} id ID of element
 */
function jump(id) {
    var element = document.getElementById(id);
    element.scrollIntoView(true);
    window.scrollBy(0, -80); // Adjust scrolling with a negative value, to offset from the top. This was requested by Anke.
}

/**
 * Detects if a comment is currently open
 */
function aCommentIsOpen() {
    var comments = document.getElementsByClassName("coFindComment");
    if (comments.length > 0) {
        for (comment of comments) {
            if (comment.hasAttribute("data-cofind-state")) {
                var state = comment.getAttribute("data-cofind-state");
                if (state != undefined) {
                    if (state == "open") {
                        return true;
                    }
                }
            }
        }
    }
    return false;
}

function firstSocketRefresh() {
    var message = {
        type: "socket",
        data: {
            socketType: "refresh"
        }
    }
    sendMessageToBackend(message); //socket events are handled in the background script
}


/**
 * generates a UUID for temporary notes 
 * @returns UUID
 * */
function uuidv4() {
    return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function (c) {
        var r = Math.random() * 16 | 0, v = c == 'x' ? r : (r & 0x3 | 0x8);
        return v.toString(16);
    });
  }

  function isThisATouchDevice() {
    var prefixes = ' -webkit- -moz- -o- -ms- '.split(' ');
    var mq = function (query) {
        return window.matchMedia(query).matches;
    }
    if (('ontouchstart' in window) || window.DocumentTouch && document instanceof DocumentTouch) {
        return true;
    }
    // include the 'heartz' as a way to have a non matching MQ to help terminate the join
    // https://git.io/vznFH
    var query = ['(', prefixes.join('touch-enabled),('), 'heartz', ')'].join('');
    return mq(query);
  }
